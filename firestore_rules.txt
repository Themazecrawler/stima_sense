rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read/write reports
    match /reports/{reportId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == resource.data.userId;
      allow update: if request.auth != null && request.auth.uid == resource.data.userId;
      allow delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Allow authenticated users to read/write likes
    match /reports/{reportId}/likes/{likeId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to read/write comments
    match /reports/{reportId}/comments/{commentId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == resource.data.userId;
      allow update: if request.auth != null && request.auth.uid == resource.data.userId;
      allow delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Allow authenticated users to read/write outages
    match /outages/{outageId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // Allow authenticated users to read/write ML training data
    match /ml_training_data/{dataId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to read/write retraining schedules
    match /retraining_schedules/{scheduleId} {
      allow read, write: if request.auth != null;
    }
  }
}
